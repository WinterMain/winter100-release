жҖҺж ·и®ҫи®Ўи§ЈејҖж•°зӢ¬жёёжҲҸ

Winter2017-09-06

еҫҲж—©д№ӢеүҚеӨ§жҰӮжҳҜ2014е№ҙзҡ„ж—¶еҖҷпјҢжҲ‘з”ЁWPFе°ұжҳҜC#е•ҰпјҢеҶҷдәҶдёҖдёӘж•°зӢ¬жёёжҲҸпјҢйӮЈж—¶д№ӢжүҖд»ҘжңүиҝҷдёӘжғіжі•пјҢдё»иҰҒиҝҳжҳҜеӣ дёәжӣҙж—©д№ӢеүҚзҺ©ж•°зӢ¬жёёжҲҸж—¶еҹәжң¬жІЎжңүжҠҠйўҳзӣ®и§ЈеҮәпјҢжңүзӮ№е°ҸеҸ—дјӨпјҢдҪҶжң¬и‘ӣи‘ӣжҜ•з«ҹжҳҜзЁӢеәҸе‘ҳе•ҠпјҢжҲ‘и§ЈдёҚеҮәпјҢеҸҜд»Ҙз”ЁзЁӢеәҸи§ЈеҮәеҳӣгҖӮжңҖиҝ‘еҸҲз”ЁJavaScriptжҠҠж•°зӢ¬жёёжҲҸзҡ„и§Јжі•еҸҲе®һзҺ°дәҶдёҖйҒҚпјҢдәҺжҳҜжғіи®°еҪ•дёӢжқҘпјҢеӨ§е®¶еҸҜд»ҘдёҖиө·жҺўи®Ё

 

йӮЈиҝҷдёӘж•°зӢ¬з®—жі•иҰҒжҖҺд№Ҳи®ҫи®Ўе‘ўпјҢжҲ‘е°Ҷж•ҙдёӘж•°зӢ¬еҲҶжҲҗдәҶд»ҘдёӢеҮ дёӘйғЁеҲҶгҖӮйҰ–е…ҲжҲ‘д»¬йңҖиҰҒе»әйҖ еҹәжң¬зҡ„з»“жһ„дҪ“пјҡ

1гҖҒе»әдёҖдёӘ9*9еӨ§е°Ҹзҡ„дәҢз»ҙж•°з»„пјҢз”ЁдәҺеӯҳеӮЁж•°зӢ¬еҗ„дёӘиҠӮзӮ№зҡ„еҖјBrickNodeгҖӮ 
2гҖҒеҲӣе»әж•°зӢ¬еҚ•дёӘиҠӮзӮ№зҡ„ж•°жҚ®з»“жһ„дҪ“BrickNodeпјҢе®ғеҢ…еҗ«дёӨдёӘдҝЎжҒҜnumberпјҲеҖјпјҢеҲқе§ӢеҖјдёә0пјүе’ҢkindпјҲз§Қзұ»пјҢжҳҜеҺҹе§Ӣзҡ„еҖјдҫӢеҰӮдёҠеӣҫдёӯж©ҷиүІзҡ„з§Қзұ»пјҢз”ЁжҲ·зј–иҫ‘зҡ„еҖјеҰӮдёҠеӣҫзҡ„з»ҝиүІз§Қзұ»пјү,дҫӢеҰӮе·ҰдёҠи§’зҡ„еҺҹе§ӢеҖј2зҡ„иЎЁзӨәдёәnumber=2пјҢkind=1 
3гҖҒиЎЁзӨәеқҗж Үзҡ„з»“жһ„дҪ“MsPointпјҢе®ғеҢ…еҗ«дёӨдёӘеҖјxе’ҢyпјҢдҫӢеҰӮе·ҰдёҠи§’зҡ„2зҡ„еқҗж ҮеҖјдёәx=0,y=0

жҺҘдёӢжқҘе°ұжҳҜжҲ‘д»¬дё»иҰҒзҡ„з®—жі•и®ҫи®ЎпјҢйӮЈиҝҷдёӘз®—жі•иҰҒжҖҺд№Ҳи®ҫи®Ўе‘ўпјҹжҲ‘зҡ„з¬¬дёҖжғіжі•жҳҜдҪҝз”ЁйҖ’еҪ’пјҢиҝҷдёӘдё»иҰҒзҡ„з®—жі•иҰҒеҒҡзҡ„дәӢжғ…е°ұжҳҜпјҡ

1гҖҒеҲӨж–ӯжҳҜеҗҰе·Із»Ҹеҫ—еҲ°дәҶзӯ”жЎҲпјҢеҰӮжһңеҫ—еҲ°еҲҷз»“жқҹпјҢеҗҰеҲҷз»§з»ӯ 
2гҖҒж·ұеәҰе…ӢйҡҶдёҖд»ҪеҪ“еүҚзҡ„ж•°зӢ¬ 
3гҖҒеҲӨж–ӯеҪ“еүҚзҡ„еқҗж ҮжҳҜеҗҰжҳҜжңҖеҗҺдёҖдёӘиҠӮзӮ№пјҢдё”numberеҖјдёҚзӯүдәҺ0еҚіе·Із»ҸеЎ«е……дәҶдёҖдёӘжңүж•ҲеҖјгҖӮеҰӮжһңж»Ўи¶іиҝҷдёӨдёӘжқЎд»¶пјҢеҲҷиЎЁзӨәж•°зӢ¬е·Із»Ҹи®Ўз®—е®ҢпјҢдё”еҫ—еҲ°дәҶзӯ”жЎҲгҖӮеҰӮжһңдёҚж»Ўи¶іеҲҷз»§з»ӯ 
4гҖҒеҲӨж–ӯеҪ“еүҚзҡ„еқҗж ҮдёҠиҠӮзӮ№жҳҜеҗҰжҳҜеҺҹе§ӢиҠӮзӮ№гҖӮеҰӮжһңжҳҜзҡ„иҜқпјҢеҲҷиҺ·еҸ–дёӢдёҖдёӘеқҗж ҮпјҢеҶҚд»Һ1йҮҚж–°ејҖе§ӢпјҢеҚійҖ’еҪ’ж–№жі•пјӣеҰӮжһңдёҚжҳҜзҡ„иҜқеҲҷиҝӣиЎҢз¬¬5жӯҘж“ҚдҪңгҖӮ 
5гҖҒиҺ·еҸ–еҪ“еүҚеқҗж ҮеҸҜеЎ«е……зҡ„ж•°еӯ—зҡ„ж•°з»„пјҢеҰӮжһңиҜҘж•°з»„зҡ„й•ҝеәҰдёә0пјҢеҲҷиЎЁзӨәеҪ“еүҚеқҗж ҮжІЎжңүд»»дҪ•ж•°еӯ—еҸҜеЎ«е……пјҢеҚіж— и§ЈпјҢзӣҙжҺҘreturnпјҢз»“жқҹеҪ“еүҚеҲҶж”Ҝзҡ„йҖ’еҪ’пјӣеҰӮжһңеҪ“еүҚж•°з»„й•ҝеәҰдёҚдёәз©әпјҢеҲҷйҒҚеҺҶж•°з»„пјҢеҸ–йҒҚеҺҶзҡ„еҖје°Ҷе…¶еЎ«е……еҲ°еҪ“еүҚж•°зӢ¬ж•°з»„пјҢеҲӨж–ӯеҪ“еүҚзҡ„еқҗж ҮжҳҜеҗҰжҳҜжңҖеҗҺдёҖдёӘиҠӮзӮ№дё”numberеҖјдёҚзӯүдәҺ0пјҢжҳҜзҡ„иҜқеҲҷиЎЁзӨәе·Із»Ҹи®Ўз®—е®ҢдәҶпјҢи®ҫзҪ®ж Үи®°жҳҜеҗҰе·Із»ҸеҸ–еҲ°зӯ”жЎҲзҡ„еҸҳйҮҸеҖјдёәtrueпјҢеҚіе·Із»Ҹи®Ўз®—е®ҢдәҶпјҢе…¶е®ғйҖ’еҪ’еҲҶж”ҜйғҪдёҚз”Ёи®Ўз®—дәҶпјҢеҰӮжһңдёҚж»Ўи¶іжҳҜжңҖеҗҺдёҖдёӘиҠӮзӮ№пјҢеҲҷиҺ·еҸ–дёӢдёҖдёӘеқҗж ҮпјҢд»Һз¬¬1жӯҘеҶҚз»§з»ӯйҖ’еҪ’гҖӮ

е…·дҪ“д»Јз Ғе®һзҺ°еҰӮдёӢжүҖзӨәпјҡ

    calculating: function (sudokuParam, point) {
        if (this.gotAnswer || this.stop){
            return;
        }

        //ж·ұеәҰе…ӢйҡҶ
        var sudoku = sudokuParam.clone();

        //еҲӨж–ӯеҪ“еүҚзҡ„еқҗж ҮжҳҜеҗҰжҳҜжңҖеҗҺдёҖдёӘиҠӮзӮ№пјҢдё”numberеҖјдёҚзӯүдәҺ0пјҲеҚіжҳҜеҗҰе·Із»ҸеЎ«е……дәҶдёҖдёӘжңүж•ҲеҖјпјү
        if (point.X == 8 && point.Y == 8 && sudoku[point.X][point.Y].number != 0) {
            //еҫ—еҲ°зӯ”жЎҲдәҶ
            this.answer = sudoku;
            //ж Үи®°е·Із»Ҹеҫ—еҲ°зӯ”жЎҲдәҶ
            this.gotAnswer = true;

            return;
        }

        //еҲӨж–ӯеҪ“еүҚиҠӮзӮ№зҡ„з§Қзұ»жҳҜеҗҰжҳҜеҺҹе§ӢеҖјпјҲеҚіж•°зӢ¬йўҳз»ҷеҮәзҡ„еҹәжң¬еҖјпјҢеҰӮдёҠеӣҫзҡ„ж©ҷиүІж–№еқ—пјү
        if (sudoku[point.X][point.Y].kind == BrickKind.Original){
            //ж №жҚ®еҪ“еүҚж•°зӢ¬еҸҠеҪ“еүҚеқҗж Үзҡ„еҖјпјҢиҺ·еҸ–дёӢдёӘеқҗж ҮпјҢе№¶з»§з»ӯи®Ўз®—
            this.calculating(sudoku, this.getNextPoint(sudoku, point));
        }
        else
        {
            //иҺ·еҸ–еҪ“еүҚеқҗж ҮеҸҜеЎ«е……е…Ҙзҡ„ж•°еӯ—з»“жһ„дҪ“пјҢз»“жһңжҳҜдёҖдёӘж•°з»„
            var brickNodes = this.getAvailableBrickNodes(sudoku, point);

            //ж•°з»„й•ҝеәҰдёә0пјҢеҚіеҪ“еүҚеҲҶж”ҜдёӢзҡ„ж•°з»„ж— и§ЈпјҢзӣҙжҺҘйҖҖеҮәеҪ“еүҚйҖ’еҪ’еҲҶж”Ҝ
            if (brickNodes.count == 0){
                return;
            }

            // еҰӮжһңеҪ“еүҚж•°з»„й•ҝеәҰдёҚдёәз©әпјҢеҲҷйҒҚеҺҶж•°з»„
            for (var i = 0 ; i < brickNodes.length; i++) {
                var brickNode = brickNodes[i];
                //еҸ–йҒҚеҺҶзҡ„еҖје°Ҷе…¶еЎ«е……еҲ°еҪ“еүҚж•°зӢ¬ж•°з»„зҡ„еҪ“еүҚеқҗж ҮдёӢ
                sudoku[point.X][point.Y] = brickNode;
                //еҲӨж–ӯеҪ“еүҚзҡ„еқҗж ҮжҳҜеҗҰжҳҜжңҖеҗҺдёҖдёӘиҠӮзӮ№дё”numberеҖјдёҚзӯүдәҺ0пјҢжҳҜзҡ„иҜқеҲҷиЎЁзӨәе·Із»Ҹи®Ўз®—е®ҢдәҶ
                if (point.X == 8 && point.Y == 8 && sudoku[8][8].number != 0) {
                    this.answer = sudoku;
                    //и®ҫзҪ®ж Үи®°жҳҜеҗҰе·Із»ҸеҸ–еҲ°зӯ”жЎҲзҡ„еҸҳйҮҸеҖјдёәtrueпјҢеҚіе·Із»Ҹи®Ўз®—е®ҢдәҶпјҢе…¶е®ғйҖ’еҪ’еҲҶж”ҜйғҪдёҚз”Ёи®Ўз®—дәҶ
                    this.gotAnswer = true;

                    return;
                }
                //еҰӮжһңдёҚж»Ўи¶іжҳҜжңҖеҗҺдёҖдёӘиҠӮзӮ№пјҢеҲҷиҺ·еҸ–дёӢдёҖдёӘеқҗж ҮпјҢд»Һз¬¬1жӯҘеҶҚз»§з»ӯйҖ’еҪ’
                this.calculating(sudoku, this.getNextPoint(sudoku, point));
            }
        }
    }

иҝҷйҮҢйқўз”ЁеҲ°дёӨдёӘж–№жі•пјҢиҝҷдёӨдёӘж–№жі•еңЁиҝҷйҮҢеҸӘйҳҗиҝ°е…·дҪ“зҡ„з”ЁеӨ„е’Ңи§ЈеҶізҡ„жҖқи·ҜпјҢе…¶е®ғеҲҷдёҚеҶҚиөҳиҝ°гҖӮ

1гҖҒgetAvailableBrickNodes(sudoku, point)дҪңз”ЁжҳҜиҺ·еҸ–еҪ“еүҚеқҗж ҮеҸҜеЎ«е…Ҙзҡ„ж•°еӯ—пјҢеҚіеҲӨж–ӯз«–жҺ’1-9жңүе“Әдәӣж•°еӯ—еҸҜз”ЁпјҢжЁӘжҺ’1-9жңүе“Әдәӣж•°еӯ—еҸҜз”ЁпјҢж–№ж јеҶ…1-9жңүе“Әдәӣж•°еӯ—еҸҜз”ЁпјҢеҸ–дёүж–№зҡ„е№¶йӣҶгҖӮз¬¬дёҖдёӘеҸӮж•°sudokuдёәеҪ“еүҚж•°зӢ¬зҡ„дәҢз»ҙж•°з»„пјҢpointжҢҮзҡ„жҳҜеҪ“еүҚеқҗж ҮеҖј 
2гҖҒgetNextPoint(sudoku, point)дҪңз”ЁжҳҜиҺ·еҸ–дёӢдёҖдёӘеқҗж Үзҡ„еҖјпјҢиҝҷдёӘж–№жі•жҜ”иҫғз®ҖеҚ•пјҢдё»иҰҒзӮ№еңЁдәҺеҲӨж–ӯеҪ“еүҚзҡ„xжҳҜеҗҰе·Із»ҸжҳҜеҲ°8дәҶпјҢеҰӮжһңеҲ°дәҶеҲҷxиҰҒеҠ 1иҖҢyиҰҒе……еҖјжңӘ0пјҢеҹәдәҺиҝҷдёҖзӮ№пјҢиҝҳиҰҒеҲӨж–ӯеҪ“еүҚзӮ№жҳҜеҗҰжҳҜеҺҹе§ӢзӮ№пјҢжҳҜеҺҹе§ӢзӮ№зҡ„иҜқиҝҳиҰҒдёҖзӣҙеүҚиҝӣпјҢзӣҙеҲ°йҒҮеҲ°йқһеҺҹе§ӢзӮ№пјҢз„¶еҗҺе°ҶиҜҘзӮ№еқҗж ҮдҪңдёәиҝ”еӣһеҖјиҝ”еӣһгҖӮеҗҢж ·з¬¬дёҖдёӘеҸӮж•°sudokuдёәеҪ“еүҚж•°зӢ¬зҡ„дәҢз»ҙж•°з»„пјҢpointжҢҮзҡ„жҳҜеҪ“еүҚеқҗж ҮеҖј
